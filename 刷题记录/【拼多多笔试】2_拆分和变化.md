## Solution
```c++
#include<iostream>
#include<string>
using namespace std;

int sum(string str)
{
	if (str.size() == 1)return 1;
	if (str.back() == '0' && str.front() == '0')return 0;
	if (str.front() == '0' || str.back() == '0')return 1;
	return str.size();
}
int main()
{
	string str;
	cin >> str;
	int res = 0;
	for (int i = 1; i<str.size(); i++)
	{
		string left = str.substr(0, i);
		string right = str.substr(i);
		res += sum(left) * sum(right);
	}
	cout << res << endl;
	return 0;
}
```
